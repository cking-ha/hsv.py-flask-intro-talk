Building a RESTful UNIX Pipe

Desgin
  1. Landing page at /
  2. Endpoint at /pipe
  3. Queues at /pipe/:queue_name
  4. These slides at /slides
RESTful reciept of data
@app.route('/pipe', methods=['POST'])
@app.route('/pipe/<queue>', methods=['POST'])
def pipe(queue='default'):
    json = request.get_json()
    content = json.get('content')

    response = pipe_message(content, queue)

    if response is not False:
        return jsonify({'status': 'ok', 'message': response})
    else:
        return jsonify({'status': 'error'})
Pipe Data somewhere
def pipe_message(message, queue):
    connection = pika.BlockingConnection(
        pika.URLParameters(app.config['RABBITMQ_URI'])
    )
    channel = connection.channel()

    channel.queue_declare(queue=queue, durable=True)

    channel.basic_publish(exchange='', 
    			  routing_key=queue_name,
			  body=message,
                          properties=pika.BasicProperties(delivery_mode=2)
			 )
    connection.close()
